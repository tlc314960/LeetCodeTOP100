public class Solution {
    public boolean hasCycle(ListNode head) {
        if(head == null || head.next == null)
            return false;
        ListNode fast = head.next, slow = head;
        while(fast != null && fast.next != null){
            if(fast == slow)
                return true;
            else{
                fast = fast.next.next;
                slow = slow.next;
            }
        }
        return false;
    }
}
